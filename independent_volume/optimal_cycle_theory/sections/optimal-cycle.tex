\section{The optimal cycle}\label{sec:cycle}

The results of Sections~1--8 provide the three stages of a closed algebraic loop:
\[
M = (\Rpos, \times)
\xrightarrow{\;\log\;}
A = (\R, +)
\xrightarrow{\;\text{Fourier}\;}
\widehat{A} = (\Z, +)
\xrightarrow{\;\exp\;}
M.
\]
We now formalize this loop and its obstruction to closure.

\begin{definition}[The additive-multiplicative cycle]\label{def:cycle}
Let $f \in L^1(\bT)$ with $f > 0$ a.e.\ and $\log f \in L^1(\bT)$. Write
\[
c_k = \int_0^1 (\log f(\theta))\, e^{-2\pi i k\theta}\, d\theta, \qquad k \in \Z,
\]
for the Fourier coefficients of $\log f$. The \emph{additive-multiplicative cycle} of $f$ consists of:
\begin{enumerate}[label=(\roman*)]
\item The \emph{DC component}: $c_0 = \int_0^1 \log f(\theta)\, d\theta$, yielding the geometric mean $G(f) = e^{c_0}$.
\item The \emph{AC component}: $\{c_k\}_{k \neq 0}$, yielding the \emph{holonomy}
\begin{equation}\label{eq:holonomy}
H(f) = \sum_{k=1}^{\infty} k\, |c_k|^2 \in [0, \infty].
\end{equation}
\item The \emph{Szeg\H{o} constant}: $E(f) = \exp(H(f)) \in [1, \infty]$.
\end{enumerate}
The holonomy $H(f)$ is the one-sided $H^{1/2}(\bT)$ Sobolev seminorm of $\log f$ (summing over positive frequencies; for real-valued $\log f$, this equals $\frac{1}{2}\|\log f\|_{H^{1/2}(\bT)}^2$) and measures the cost of traversing the cycle.
\end{definition}

\begin{theorem}[Holonomy trichotomy]\label{thm:holonomy-trichotomy}
Let $f$ be as in \cref{def:cycle}. Then exactly one of the following holds:
\begin{enumerate}[label=(\roman*)]
\item $H(f) = 0$, equivalently $E(f) = 1$, if and only if $c_k = 0$ for all $k \geq 1$, if and only if $f$ is constant a.e.\ (the Gaussian spectral density). The cycle closes perfectly.
\item $0 < H(f) < \infty$, equivalently $1 < E(f) < \infty$: the Strong Szeg\H{o} class. The cycle is non-trivial with finite cost, and $D_n(f) \sim G(f)^n \cdot E(f)$.
\item $H(f) = \infty$, equivalently $E(f) = \infty$: the cycle breaks. The ratio $D_n(f) / G(f)^n$ diverges or oscillates without a finite limit.
\end{enumerate}
\end{theorem}

\begin{proof}
Part~(i) is immediate from the definition: $H(f) = 0$ requires $c_k = 0$ for all $k \geq 1$, so $\log f = c_0$ is constant, hence $f = e^{c_0}$ is constant. Parts~(ii) and~(iii) are the content of the Strong Szeg\H{o} Limit Theorem (\cref{thm:szego} in \S\ref{sec:hardy}).
\end{proof}

\begin{remark}[The Gaussian as unique fixed point]
The Gaussian spectral density $f \equiv \sigma^2$ is simultaneously:
\begin{enumerate}[label=(\alph*)]
\item the Fourier eigenfunction (\cref{thm:gaussian-fixed}),
\item the heat kernel (\cref{thm:heat-semigroup}),
\item the unique outer function seed ($c_k = 0$ for $k \neq 0$),
\item the unique spectral density with $E(f) = 1$ (\cref{thm:holonomy-trichotomy}(i)).
\end{enumerate}
These four characterizations are manifestations of a single fact: the Gaussian is the fixed point of the optimal cycle.
\end{remark}

\begin{definition}[Constrained optimal cycle]\label{def:optimal-cycle}
Let $\Omega$ denote the space of spectral densities $f \in L^1(\bT)$ with $f > 0$ a.e.\ and $\log f \in L^1(\bT)$. For each $t \geq 0$, define the \emph{feasible set at lag $t$}:
\[
\Omega_t = \bigl\{f \in \Omega : \widehat{f}(k) = r_k \text{ for } |k| \leq t\bigr\},
\]
where $r_0, r_1, \ldots$ are given autocovariance values (observed data). The \emph{constrained holonomy} is
\[
H_t^* = \inf_{f \in \Omega_t} H(f).
\]
The minimizer $f_t^*$, when it exists, is the spectral density of minimum holonomy compatible with the observations.
\end{definition}

\begin{proposition}[Monotonicity of the constrained cycle]\label{prop:holonomy-monotone}
In the notation of \cref{def:optimal-cycle}:
\begin{enumerate}[label=(\roman*)]
\item \textbf{Shrinking feasible sets}: $\Omega_0 \supseteq \Omega_1 \supseteq \Omega_2 \supseteq \cdots$.
\item \textbf{Monotone holonomy}: $H_0^* \leq H_1^* \leq \cdots \leq H(f_{\mathrm{true}})$, with equality in the limit if $f_{\mathrm{true}} \in \Omega$.
\item \textbf{Maximum-entropy minimizer}: the minimizer $f_t^*$ is the maximum-entropy spectral density compatible with $r_0, \ldots, r_t$, i.e., the autoregressive AR$(t)$ spectral density (Burg's theorem \cite{Burg}).
\item \textbf{Dual convergence}: the prediction error $\sigma_t^2 = \exp\bigl(\int_{\bT} \log f_t^*\bigr)$ decreases monotonically to $\sigma_{\mathrm{true}}^2$, while $H_t^*$ increases monotonically to $H(f_{\mathrm{true}})$---convergence from opposite directions.
\end{enumerate}
\end{proposition}

\begin{proof}
Part~(i): each additional constraint $\widehat{f}(t+1) = r_{t+1}$ restricts the feasible set.

Part~(ii): immediate from~(i), since the infimum over a smaller set is at least as large.

Part~(iii): by the Szeg\H{o}--Verblunsky identity (\cref{prop:verblunsky}), the Verblunsky coefficients $\alpha_0, \ldots, \alpha_{t-1}$ are uniquely determined by the autocovariances $r_0, \ldots, r_t$ via the Levinson recursion. They are the same for all $f \in \Omega_t$. The AR$(t)$ model is characterized by $\alpha_k = 0$ for all $k \geq t$. Split $H(f)$ using \cref{prop:verblunsky}(iii):
\[
H(f) = \underbrace{-\sum_{k=1}^{t-1} k\log(1-|\alpha_k|^2)}_{\text{constant on }\Omega_t} \;+\; \underbrace{\left(-\sum_{k=t}^{\infty} k\log(1-|\alpha_k|^2)\right)}_{\geq\, 0,\;\; =\, 0 \;\text{iff}\; \alpha_k = 0\;\forall\, k \geq t}.
\]
Similarly, for the geometric mean (\cref{prop:verblunsky}(ii)):
\[
\log G(f) = \underbrace{\log r_0 + \sum_{k=0}^{t-1}\log(1-|\alpha_k|^2)}_{\text{constant on }\Omega_t} \;+\; \underbrace{\sum_{k=t}^{\infty}\log(1-|\alpha_k|^2)}_{\leq\, 0,\;\; =\, 0 \;\text{iff}\; \alpha_k = 0\;\forall\, k \geq t}.
\]
Since $-\log(1-|\alpha_k|^2) \geq 0$ with equality iff $\alpha_k = 0$: $H(f) \geq H(f_{\mathrm{AR}(t)})$ and $G(f) \leq G(f_{\mathrm{AR}(t)})$, with equality in both iff $\alpha_k = 0$ for all $k \geq t$, i.e., $f = f_{\mathrm{AR}(t)}$. This is Burg's theorem \cite{Burg}: the AR$(t)$ density simultaneously minimizes holonomy and maximizes the geometric mean (entropy) on $\Omega_t$.

Part~(iv): since $f_t^* = f_{\mathrm{AR}(t)}$, we have $\sigma_t^2 = G(f_t^*)$ and $H_t^* = H(f_t^*)$. As $t$ increases: each additional Verblunsky coefficient $\alpha_t \neq 0$ (generically) adds a positive term to $H$ and a negative term to $\log G$, so $H_t^*$ increases and $\sigma_t^2$ decreases monotonically. These are dual aspects of the same convergence: the Verblunsky coefficients apportion the true spectral density's holonomy into constrained (known) and free (unknown) parts, and each observation converts one coefficient from free to constrained.
\end{proof}

\begin{remark}[Connection to GPI]
The filtration $\mathcal{G}_t = \sigma(r_0, \ldots, r_t)$ generated by observed autocovariances is a functor in $\Proj_Y$ (\cref{def:gpi-category}). The constrained holonomy $H_t^*$ is the evaluation of this functor at the holonomy functional, and its monotonicity (\cref{prop:holonomy-monotone}(ii)) is an instance of the residual monotonicity of \cref{cor:monotonicity}. The dual convergence---prediction error decreasing, holonomy increasing---is the content of the in-context GPI theorem (\cref{thm:in-context-gpi}) applied to the spectral density space.
\end{remark}

\begin{remark}[Measurement, estimation, and prediction]\label{rem:three-layers}
The re-cycle operates through three layers, each formalized by the preceding results.

\emph{Layer~1: Measurement.}
Build the $\sigma$-algebra $\mathcal{G}_t = \sigma(r_0, \ldots, r_t)$ from observed autocovariances. Without measurement, $\mathcal{G} = \{\varnothing, \Omega\}$, and by \cref{prop:cond-exp-optimal}:
\[
\mathbb{E}[Y \mid \mathcal{G}] = \mathbb{E}[Y]
\]
---the global mean, maximum ignorance. In cycle language: the feasible set $\Omega_0$ is unconstrained, so $H_0^* = 0$ trivially---not because the true density is Gaussian, but because no data constrains the minimizer away from the constant density. The prediction is pure DC: $G(f)$ without $E(f)$.

\emph{Layer~2: Estimation (re-cycle).}
Each measurement refines $\mathcal{G}_t$. By \cref{cor:monotonicity}, the residual $R_Y(\mathcal{G}_t)$ decreases monotonically. In spectral language: $f_t^*$ (the AR$(t)$ minimizer of \cref{prop:holonomy-monotone}) sharpens, $\nu$ increases, $t_\nu$ thins. This is slow but guaranteed: $O(1/\sqrt{n})$ by the CLT rate.

\emph{Layer~3: Prediction (direct computation).}
Given the estimated spectral density $f_t^*$, compute consequences exactly---the forward problem on known structure. This layer is only as good as Layer~2: prediction inherits the estimation uncertainty, but the forward computation is exact, so input uncertainty maps to output uncertainty in a fully characterized way.

The closed operational loop
\begin{gather*}
\text{Measure}
\xrightarrow{\;\text{re-cycle}\;}
\text{Estimate}
\xrightarrow{\;\text{direct computation}\;}
\text{Predict} \\
\xrightarrow{\;\text{target next observation}\;}
\text{Measure better}
\end{gather*}
is the optimal cycle applied operationally. Each revolution refines Layer~2: $\nu$ increases, $H$ decreases.
\end{remark}

\begin{example}[The Student $t$-distribution as canonical re-cycle]\label{ex:student-t}
\emph{Student $t$ with $\nu$ degrees of freedom is the re-cycle snapshot at observation $\nu$.}

The Student $t_\nu$ family parameterizes the entire re-cycle trajectory with a single real parameter $\nu \in [1, \infty]$:
{\small
\[
\renewcommand{\arraystretch}{1.4}
\begin{array}{cclcl}
\hline
\nu & \text{Distribution} & H(f_{t_\nu}) & E(f_{t_\nu}) & \text{Cycle state} \\
\hline
\nu = 1 & \text{Cauchy} & \infty & \infty & \text{Broken} \\
1 < \nu \leq 2 & \text{Heavy-tailed } t & \infty & \infty & E[X^2] = \infty \\
2 < \nu < \infty & \text{Moderate } t & \text{finite}, > 0 & > 1 & \text{Non-trivial---learning} \\
\nu \to \infty & \text{Gaussian} & 0 & 1 & \text{Fixed point} \\
\hline
\end{array}
\]
}\end{example}

\begin{proposition}[Student $t$ convergence along the re-cycle]\label{prop:student-convergence}
For $\nu > 0$, define the \emph{spectral density associated to the Student $t_\nu$ distribution} as the periodized autocovariance:
\[
f_{t_\nu}(\theta) = \sum_{k \in \Z} r_k\, e^{2\pi ik\theta}, \qquad r_k = \varphi_{t_\nu}(2\pi k),
\]
where $\varphi_{t_\nu}(\xi) = \frac{2^{1-\nu/2}}{\Gamma(\nu/2)}|\sqrt{\nu}\,\xi|^{\nu/2}K_{\nu/2}(\sqrt{\nu}\,|\xi|)$ is the characteristic function of the $t_\nu$ distribution and $K_{\nu/2}$ is the modified Bessel function of the second kind. For $\nu > 2$, the second moment $E[X^2] = \nu/(\nu-2)$ is finite, $r_k$ decays exponentially in $|k|$, and $f_{t_\nu} > 0$ with $\log f_{t_\nu} \in L^1(\bT)$.
\begin{enumerate}[label=(\roman*)]
\item The critical value is $\nu^* = 2$: the Szeg\H{o} constant satisfies $E(f_{t_\nu}) < \infty$ if and only if $\nu > 2$, corresponding to the threshold where the second moment $E[X^2]$ becomes finite and the autocovariance is well-defined and summable.
\item For $\nu > 2$: $H(f_{t_\nu}) = O(\log\nu / \nu)$ as $\nu \to \infty$. The convergence rate to the Gaussian fixed point is algebraic.
\item The unique minimizer of $H$ over the family $\{f_{t_\nu} : \nu > 2\}$ is $\nu = \infty$ (Gaussian), confirming the cycle's unique fixed point.
\end{enumerate}
\end{proposition}

\begin{proof}[Proof sketch]
(i) For $\nu \leq 2$, the second moment diverges ($E[X^2] = \nu/(\nu-2) \to \infty$ as $\nu \to 2^+$), so $r_0 = \infty$ and the spectral density is not in $L^1$. For $\nu > 2$, the exponential decay of $K_{\nu/2}$ ensures $r_k = O(e^{-c|k|})$ for some $c > 0$, so $f_{t_\nu}$ is smooth and strictly positive, giving $\log f_{t_\nu} \in H^{1/2}(\bT)$ and $E(f_{t_\nu}) < \infty$.

(ii) By the Szeg\H{o}--Verblunsky identity (\cref{prop:verblunsky}), $H(f_{t_\nu}) = -\sum_{k=1}^{\infty}k\log(1-|\alpha_k(\nu)|^2)$. For large $\nu$, the $t_\nu$ characteristic function satisfies $\varphi_{t_\nu}(\xi) = e^{-\xi^2/2}(1 + O(\xi^4/\nu))$, so the spectral density approaches the Gaussian at rate $O(1/\nu)$ uniformly. The Verblunsky coefficients inherit this: $|\alpha_k(\nu)|^2 = O(k^{-2}\nu^{-1})$, giving $H(f_{t_\nu}) \approx \sum_{k=1}^{N} k|\alpha_k|^2 = O(\nu^{-1})\sum_{k=1}^{N} k^{-1} = O(\log\nu/\nu)$ where $N \sim \nu$ is the effective cutoff beyond which $|\alpha_k|$ is negligible.

(iii) By \cref{prop:verblunsky}(iii), $H(f_{t_\nu}) > 0$ for any $\nu < \infty$ (since some $\alpha_k \neq 0$), and $H(f_{t_\infty}) = 0$ (all $\alpha_k = 0$, Gaussian).
\end{proof}

\begin{remark}[Canonicality]
The Student $t$ family is canonical because it parameterizes the \emph{entire} re-cycle trajectory with a single real parameter $\nu \in [1, \infty]$. Every other natural family (tempered stable, variance-gamma, generalized hyperbolic) traverses only part of the trichotomy. The $t$-family is the geodesic through holonomy space.
\end{remark}

\begin{remark}[Only Gauss sees the Gaussian]\label{rem:only-gauss}
This remark has three layers of meaning.

\emph{Historical.} Gauss understood this object; others used it without seeing it. The name ``normal distribution''---never Gauss's term---encodes the false belief that the Gaussian is the \emph{common case}. It is not. It is the \emph{limit}.

\emph{Mathematical.} $E(f) = 1$ if and only if $f$ is Gaussian (\cref{thm:holonomy-trichotomy}). Only the Gaussian system has zero holonomy. Only the Gaussian sees itself perfectly through the cycle. Every other system incurs loss.

\emph{Epistemological.} $E(f_{t_\nu}) = 1$ requires $\nu = \infty$: the observer must \emph{become} the system. At $\nu = \infty$ you are no longer observing---you are the cycle itself. The map has become the territory.

The CLT does not say ``things converge to normal.'' It says: \emph{the re-cycle converges to the cycle}. The Student $t_\nu$ is the re-cycle at finite time. The Gaussian is the cycle itself. Every actual measurement is $t_\nu$ with some finite $\nu$. The Gaussian is what they approach but none of them are---the $c$ of probability theory: unreachable, structural, defining everything by being the limit nothing reaches.
\end{remark}

\begin{remark}[Dual ascent on $\varepsilon$: the re-cycle]
\label{rmk:dual-ascent}
The holonomy $H(f)$ determines a natural schedule for the barrier
parameter~$\varepsilon$.
Let $w^{(i)} \propto e^{-J^{(i)}/\varepsilon}$ be the Boltzmann
weights of an MPPI sample (\S\ref{sec:mppi}), and define the
weight entropy $S = -\sum_i w^{(i)} \log w^{(i)}$.
Low entropy ($S < H_{\mathrm{target}}$) signals that the weights
have collapsed to a few samples---the effective sample size is too
small---and $\varepsilon$ should increase to broaden exploration.
High entropy signals that the distribution is diffuse and
$\varepsilon$ can decrease to sharpen the barrier.
The multiplicative update rule is:
\[
  \varepsilon \;\leftarrow\;
  \begin{cases}
    \alpha_{\mathrm{up}}\,\varepsilon
      & \text{if } S < H_{\mathrm{target}}, \\[2pt]
    \alpha_{\mathrm{down}}\,\varepsilon
      & \text{otherwise},
  \end{cases}
  \qquad
  H_{\mathrm{target}} \leftarrow \gamma\, H_{\mathrm{target}},
\]
where $\alpha_{\mathrm{up}} > 1 > \alpha_{\mathrm{down}} > 0$
and $\gamma \in (0,1)$ is the annealing rate.
This is a multiplicative dual ascent: each iteration traverses
the optimal cycle once (sample in log-space, reweight
multiplicatively, update $\varepsilon$), and the schedule
$\varepsilon_n \to 0$ drives the Boltzmann distribution toward
the deterministic optimum.
The entropy threshold $H_{\mathrm{target}}$ decays geometrically,
ensuring that the effective sample size remains bounded away
from both $1$ (collapse) and $N$ (uniform), in analogy with the
Szeg\H{o} condition
$\int_\bT \log f\,d\theta > -\infty$
that separates the non-trivial basin from the degenerate boundary.
\end{remark}

\begin{proposition}[Online mirror descent interpretation]
\label{prop:omd}
Let $\mathcal{Q} = \{q_\theta : \theta \in \Theta\}$ be an exponential family of distributions over the feasible set (\cref{def:optimal-cycle}). The re-cycle (\cref{rmk:dual-ascent}) is an instance of Online Mirror Descent~\cite{Nemirovski-Yudin,Shalev-Shwartz} with:
\begin{itemize}
\item \emph{Primal space:} $\Theta$ (natural parameters of $q_\theta$).
\item \emph{Mirror map:} $\varphi(\theta) = \log Z(\theta)$ (log-partition function, i.e.\ negative entropy of $q_\theta$).
\item \emph{Bregman divergence:} $D_\varphi(\theta' \| \theta) = D_{\KL}(q_{\theta'} \| q_\theta)$.
\item \emph{Loss at round $n$:} $\ell_n(\theta) = \mathbb{E}_{q_\theta}[J(\beta)]$, estimated by $N$ Monte Carlo samples $\beta^{(i)} \sim q_\theta$.
\item \emph{Step size:} $\eta_n = \varepsilon_n$ (the barrier/temperature parameter).
\end{itemize}
The OMD update
\[
q_{n+1} = \arg\min_q \bigl\{\langle \hat{\nabla}\ell_n, q\rangle + \tfrac{1}{\varepsilon_n}\, D_{\KL}(q \,\|\, q_n)\bigr\}
\]
has closed-form solution
\[
q_{n+1}(\beta) \propto q_n(\beta)\,\exp\!\bigl(-J(\beta)/\varepsilon_n\bigr),
\]
which is exactly the Boltzmann reweighting $w^{(i)} \propto \exp(-J^{(i)}/\varepsilon_n)$ followed by moment matching (weighted mean and covariance update).
\end{proposition}

\begin{proof}[Proof sketch]
The minimizer of $\langle g, q\rangle + \frac{1}{\eta} D_{\KL}(q \| p)$ over distributions $q$ is $q^*(\beta) \propto p(\beta)\exp(-\eta\, g(\beta))$ by the Lagrange multiplier on the normalization constraint. Setting $p = q_n$, $g = J$, $\eta = 1/\varepsilon_n$ gives the result.
\end{proof}

\begin{remark}[Regret bound via information gain]
\label{rmk:omd-regret}
Model the objective landscape with a Gaussian process surrogate
$\GP(\mu_0, k)$, where $\mu_0$ is a prior mean (e.g., the cost at a
known safe gait) and
$k(\beta,\beta') = \exp\!\bigl(-\tfrac{\|\beta-\beta'\|^2}{2\ell^2}\bigr)$
is the squared-exponential kernel in dimension $d = \dim\Theta$.
The GP is a \emph{transport model}: the posterior
$q_n = \GP(\mu_n, \Sigma_n \mid \mathcal{D}_n)$ defines a conditional
flow from prior to truth (\cref{prop:kl-contract}), not a generative
claim that $J$ was sampled from the GP.
The GP-UCB acquisition function
\begin{equation}\label{eq:gp-ucb}
\textcolor{AcqColor}{
  \ell_n(\beta)
  = \mu_{n-1}(\beta)
    - \sqrt{\bar\beta_n}\;\sigma_{n-1}(\beta)
}
\end{equation}
(where $\mu_{n-1},\sigma_{n-1}$ are the posterior mean and standard
deviation after $n-1$ observations and
$\bar\beta_n = 2\log(|\Theta|\,n^2\pi^2/6\delta)$,
with $|\Theta|$ the cardinality of a finite cover of~$\Theta$)
selects $\beta_n = \arg\min_{\beta \in S_n} \ell_n(\beta)$, yielding the
cumulative regret bound~\cite{SrinivasGPUCB}:
\begin{equation}\label{eq:gp-regret}
R_T
= \sum_{n=1}^{T}\bigl[J(\beta_n) - J(\beta^*)\bigr]
\leq \cO\!\bigl(\sqrt{T\,\gamma_T\,\log T}\bigr),
\qquad
\gamma_T = \cO\!\bigl((\log T)^{d+1}\bigr).
\end{equation}
Three consequences:
\begin{enumerate}[label=(\roman*)]
\item \emph{Sub-linear regret.}
$R_T/T \to 0$ unconditionally whenever $J$ lies in the RKHS of~$k$.
No bounded-gradient assumption is needed---the information gain
$\gamma_T$ replaces it.
\item \emph{Information-gain interpretation.}
$\gamma_T$ counts the effective number of independent ``directions''
observed by iteration~$T$. For the SE kernel in dimension~$d$,
$\gamma_T$ is poly-logarithmic, so the regret is essentially
$\cO(\sqrt{T}\,(\log T)^{(d+1)/2})$.
\item \emph{Holonomy connection.}
The information gain $\gamma_T$ plays the role of the
holonomy $H(f)$: it measures how far the conditional flow
has transported the GP prior toward the truth
(\cref{prop:kl-contract}).
$R_T > 0$ iff $\gamma_T > 0$ iff the cycle has not closed.
Convergence $R_T/T \to 0$ means the holonomy vanishes
asymptotically, recovering the OMD interpretation of
\cref{prop:omd} with a sharper, information-theoretic rate.
\end{enumerate}
\end{remark}

\begin{remark}[Re-cycle algorithm]
\label{rmk:omd-algorithm}
We give the GP-UCB formulation of the re-cycle, making explicit
the acquisition-update-contraction structure that replaces
the MPPI sample-reweight loop of~\cref{prop:omd}.
\cref{alg:riccati-eval} encapsulates the inner trajectory solver;
\cref{alg:recycle} is the outer MPC loop that calls it,
with a two-phase acquisition strategy: Phase~1 evaluates
$2d{+}1$ sigma points for a deterministic GP warm start
(\cref{rmk:unscented}, Level~3); Phase~2 refines via
GP-UCB acquisition.
The six guarantee propositions that follow establish their
formal properties.
\end{remark}

\begin{algorithm}[ht]
\caption{Riccati trajectory evaluation}
\label{alg:riccati-eval}
\begin{algorithmic}[1]
\Require Gait parameter~$\beta$, $M$~domain-randomized copies
\Ensure Cost~$J$, feedback gains $K(t),\, k(t)$
\State Decode $\beta \to$ contact schedule
  $\sigma_k(t) \in [0,1]^4$ via B-spline
\State Backward solve $\to K(t),\, k(t)$
  \hfill\textcolor{ConColor}{\small Riccati or CUDP
  (\cref{rmk:unscented})}
\State Forward-simulate with $M$ copies
  \hfill\textcolor{ConColor}{\small MC or sigma points}
\State $J \gets \frac{1}{M}\sum_{m=1}^{M} J^{(m)}(\beta)$
  \hfill\textcolor{ConColor}{\small noise-averaged}
\State \Return $J,\; K(t),\; k(t)$
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[ht]
\caption{Re-cycle: GP-UCB gait optimization with
  sigma-point initialization}
\label{alg:recycle}
\begin{algorithmic}[1]
\Require Prior $\GP(\mu_0,k)$, seed point $\beta_0$ with
         $J(\beta_0) \leq c_{\max}$, dimension $d = \dim\Theta$,
         threshold $c_{\max}$, horizon~$T$
\Ensure Optimized gait $\beta^*$, posterior $(\mu_T,\sigma_T)$
\Statex \textcolor{CycleColor}{$\triangleright$
  \textbf{Phase~1} ($n \leq 2d{+}1$): sigma-point warm start;\;
  \textbf{Phase~2} ($n > 2d{+}1$): UCB refinement}
\State $\{\beta^{(j)}\}_{j=1}^{2d+1} \gets$ sigma points from
  $(\mu_0, \Sigma_0)$ in~$\Theta \cap S_1$
  \hfill\textcolor{CycleColor}{\small unscented
  (\cref{rmk:unscented}, Level~3)}
\Statex \textcolor{ConColor}{$\triangleright$
  $S_1 \gets \{\beta : \mu_0(\beta) + \sqrt{\bar\beta_1}\,\sigma_0(\beta)
  \leq c_{\max}\}$, seeded from~$\beta_0$}
\For{$n = 1, \ldots, T$}
  \Statex \hspace{\algorithmicindent}%
    \textcolor{AcqColor}{$\triangleright$ \textbf{Acquisition}}
  \State $\beta_n \gets
    \begin{cases}
      \beta^{(n)} & n \leq 2d{+}1
        \quad\textcolor{CycleColor}{\text{\small(sigma-point)}}
        \\[3pt]
      \displaystyle\arg\min_{\beta \in S_n}\bigl[
        \mu_{n-1}(\beta) - \sqrt{\bar\beta_n}\;\sigma_{n-1}(\beta)
      \bigr] & n > 2d{+}1
        \quad\textcolor{AcqColor}{\text{\small(GP-LCB)}}
    \end{cases}$
  \Statex \hspace{\algorithmicindent}%
    \textcolor{ConColor}{$\triangleright$ \textbf{Evaluate
    (\cref{alg:riccati-eval})}}
  \State $J_n,\, K_n(t),\, k_n(t) \gets
    \textsc{RiccatiEval}(\beta_n, M)$
    \hfill\textcolor{ConColor}{\small \cref{alg:riccati-eval}}
  \Statex \hspace{\algorithmicindent}%
    \textcolor{PostColor}{$\triangleright$ \textbf{Posterior update
    (Bayesian)}}
  \State $\mathcal{D}_n \gets \mathcal{D}_{n-1}
    \cup \{(\beta_n, J_n)\}$
  \State Update GP: $\mu_n, \sigma_n \mid \mathcal{D}_n$
    \hfill\textcolor{PostColor}{\small $\cO(\gamma_n^3)$ via MINI-META
    \cite{CalandrielloMINIMETA}}
  \Statex \hspace{\algorithmicindent}%
    \textcolor{ConColor}{$\triangleright$ \textbf{Safety \& contraction}}
  \State $S_{n+1} \gets \bigl\{\beta :
    \mu_n(\beta) + \sqrt{\bar\beta_n}\,\sigma_n(\beta)
    \leq c_{\max}\bigr\}$
    \hfill\textcolor{ConColor}{\small pessimistic safe set
    \cite{SuiSafeOpt}}
\EndFor
\State \Return $\beta^* = \arg\min_{n} J_n$,\;
  $(\mu_T, \sigma_T)$
\end{algorithmic}
\end{algorithm}

\textcolor{CycleColor}{\textbf{Cycle connection.}}
Each iteration of \cref{alg:recycle} traverses
$M \xrightarrow{\log} A \xrightarrow{\text{Fourier}} \hat{A}
\xrightarrow{\exp} M$ once.
Line~3 is the $\log$~map (acquisition in parameter space).
\cref{alg:riccati-eval} is the Fourier transform (inner solve evaluates
the cost functional on the current ``frequency''~$\beta_n$).
Lines~5--6 are the $\exp$~map (posterior update returns a distribution
over parameter space).
Line~7 is the entropy compression: the safe set $S_{n+1}$ contracts,
mirroring the Szeg\H{o} condition
$\int_\bT \log f\,d\theta > -\infty$ that separates the non-trivial
basin from the degenerate boundary.

\begin{proposition}[No-regret convergence]
\label{prop:gp-regret}
Suppose $J - \mu_0$ lies in the RKHS of the SE kernel with $\|J - \mu_0\|_k \leq B$.
Then \cref{alg:recycle} achieves cumulative regret
\[
R_T \leq \cO\!\bigl(\sqrt{T\,(\log T)^{d+1}\,\log T}\bigr)
= \widetilde{\cO}(\sqrt{T}),
\]
unconditionally on the smoothness of~$J$.
\end{proposition}

\begin{proof}[Proof sketch]
By \cite[Theorem~1]{SrinivasGPUCB}, the GP-UCB regret satisfies
$R_T \leq \cO(\sqrt{T\,\gamma_T\,\bar\beta_T})$ where
$\gamma_T = \cO((\log T)^{d+1})$ for the SE kernel and
$\bar\beta_T = \cO(\log T)$, giving $R_T = \widetilde{\cO}(\sqrt{T})$.
\end{proof}

\begin{proposition}[Per-step safety]
\label{prop:safe-gp}
Define the constraint $g(\beta) = J(\beta) - c_{\max}$.
Under Lipschitz continuity of~$g$ and with sigma points
drawn from~$S_1$,
\cref{alg:recycle} ensures $g(\beta_n) \leq 0$ for all $n \geq 1$
with probability at least $1-\delta$.
\end{proposition}

\begin{proof}[Proof sketch]
The safe set $S_{n+1}$ (line~7) uses the upper confidence bound
$\mathrm{ucb}_n(\beta)
= \mu_n(\beta) + \sqrt{\bar\beta_n}\,\sigma_n(\beta)
\leq c_{\max}$, so any $\beta \in S_{n+1}$ satisfies
$J(\beta) \leq c_{\max}$ w.h.p.\ by the GP confidence
interval~\cite{SuiSafeOpt,BerkenkampSafeMPC}.
\end{proof}

\begin{proposition}[KL contraction via conditional flow]
\label{prop:kl-contract}
The GP posterior sequence $q_0, q_1, \ldots$ from \cref{alg:recycle}
forms a conditional flow on the space of Gaussian measures.
Let $\pi(\beta) \propto \exp(-J(\beta)/\varepsilon)$ be the Boltzmann
target and define the per-step information ratio
\[
  \eta_n
  = \frac{\sigma^2_{n-1}(\beta_n)}
         {\sigma^2_{n-1}(\beta_n) + \sigma^2_{\mathrm{noise}}}.
\]
Then:
\begin{enumerate}[label=(\roman*)]
\item The mutual information satisfies
$I(J;\, \mathcal{D}_T) = \gamma_T
= \tfrac{1}{2}\sum_{n=1}^{T}
  \log\bigl(1 + \sigma^2_{n-1}(\beta_n)
  /\sigma^2_{\mathrm{noise}}\bigr)$.
\item The continuous-time KL contraction rate is
\[
\textcolor{ConColor}{
  \frac{d}{dt}\,D_{\KL}(q_t \| \pi)
  \leq -\frac{2}{\|k\|_{\mathrm{op}}}\,
       D_{\KL}(q_t \| \pi),
}
\]
where $\|k\|_{\mathrm{op}}$ is the operator norm of the kernel
and $1/\|k\|_{\mathrm{op}}$ is the Gaussian log-Sobolev constant
(Bakry--\'Emery~\cite{BakryEmery,Ledoux}).
\end{enumerate}
\end{proposition}

\begin{proof}[Proof sketch]
(i)~Standard GP information gain: each observation at $\beta_n$ with
noise $\sigma^2_{\mathrm{noise}}$ contributes
$\tfrac{1}{2}\log(1+\sigma^2_{n-1}(\beta_n)
/\sigma^2_{\mathrm{noise}})$
to the mutual information about~$J$.

(ii)~The GP prior $q_0 = \GP(\mu_0, k)$ is a Gaussian measure on
function space.
By the Bakry--\'Emery criterion~\cite{BakryEmery,Ledoux},
every Gaussian measure satisfies a log-Sobolev inequality with
constant $\alpha = \lambda_{\min}(k^{-1}) = 1/\|k\|_{\mathrm{op}}$.
Each posterior update $q_n \mapsto q_{n+1}$ is a rank-$1$
contraction of the covariance operator:
$\Sigma_{n+1} = \Sigma_n
  - k(\cdot,\beta_{n+1})\,k(\beta_{n+1},\cdot)
  /(\sigma^2_n(\beta_{n+1}) + \sigma^2_{\mathrm{noise}})$.
This is the Bures--Wasserstein optimal transport between
Gaussians, and the log-Sobolev inequality gives the
exponential KL decay along this conditional
flow~\cite{Ledoux}.
The observation noise $\sigma^2_{\mathrm{noise}}$ from
domain randomization (\cref{rem:domain-rand}) modulates the
per-step information through~$\eta_n$: larger noise slows
contraction but improves robustness.
\end{proof}

\begin{proposition}[Trajectory contraction]
\label{prop:riccati-contract}
Under the stabilizability conditions of~\cref{sec:riccati},
the Riccati feedback law (\cref{alg:riccati-eval}, line~2) yields
\[
\textcolor{ConColor}{
  \|x(t) - \bar{x}(t)\|_P
  \leq \rho^{T-t}\,\|x(T) - \bar{x}(T)\|_P,
}
\]
where $P$ is the Riccati cost-to-go matrix and $\rho < 1$ is the
spectral contraction rate.
\end{proposition}

\begin{proof}[Proof sketch]
Standard Riccati contraction: the value function
$V(x,t) = (x-\bar{x})^\top P(t)(x-\bar{x})$ satisfies
$V(x,t) \leq \rho^2 V(x,t+1)$ along closed-loop trajectories,
so $\|x-\bar{x}\|_P$ decays geometrically backward from the terminal
condition.
\end{proof}

\begin{proposition}[Computational scalability]
\label{prop:scalable}
Using the MINI-META sparse approximation~\cite{CalandrielloMINIMETA}
with effective dictionary size bounded by $\gamma_n = \cO((\log n)^{d+1})$,
the total computational cost of \cref{alg:recycle} over $T$ iterations
is
\[
\cO\!\bigl(T\,(\log T)^{3(d+1)}\bigr),
\]
i.e., poly-logarithmic per step.
\end{proposition}

\begin{proof}[Proof sketch]
Each GP update (line~6) costs $\cO(\gamma_n^3)$ via MINI-META, where
$\gamma_n = \cO((\log n)^{d+1})$ for the SE kernel.
Summing: $\sum_{n=1}^T \gamma_n^3 = \cO(T\,(\log T)^{3(d+1)})$.
\end{proof}

\begin{proposition}[Unconditional two-level contraction]
\label{prop:unconditional}
\cref{alg:recycle} achieves two-level contraction:
\begin{enumerate}[label=(\roman*)]
\item \emph{Outer level (GP-UCB):} the gait parameter $\beta_n$
converges to $\beta^*$ at rate $\widetilde{\cO}(1/\sqrt{n})$
(\cref{prop:gp-regret}), with per-step safety
(\cref{prop:safe-gp}).
\item \emph{Inner level (Riccati):} for each fixed~$\beta$,
the trajectory converges to the nominal at
rate~$\rho^{T-t}$ (\cref{prop:riccati-contract}).
\item \emph{Combined:} the information gain
$\gamma_T = I(J;\,\mathcal{D}_T)$ simultaneously
controls the regret
(\cref{prop:gp-regret}: $R_T \leq \cO(\sqrt{T\gamma_T\log T})$)
and the KL contraction rate
(\cref{prop:kl-contract}: exponential at rate
$2/\|k\|_{\mathrm{op}}$),
with total cost $\cO(T(\log T)^{3(d+1)})$
(\cref{prop:scalable}).
\end{enumerate}
\end{proposition}

\begin{proof}[Proof sketch]
Combine \cref{prop:gp-regret,prop:safe-gp,prop:kl-contract,%
prop:riccati-contract,prop:scalable}.
The outer loop provides global convergence with safety;
the inner loop provides local trajectory tracking;
KL contraction unifies both levels.
\end{proof}

\begin{remark}[Receding-horizon MPC interpretation]
\label{rmk:mpc}
\cref{alg:recycle} is a model-predictive controller with
state~$s_n = (\mu_n, \Sigma_n)$ (GP posterior),
action~$a_n = \beta_n$ (query point),
and dynamics~$s_{n+1} = \mathrm{BayesUpdate}(s_n, \beta_n, J_n)$.
Two classical MPC guarantees hold:
\begin{enumerate}[label=(\roman*)]
\item \emph{Recursive feasibility.}
The safe set~$S_n$ is non-empty for all $n \geq 1$.
Indeed, $\beta_0 \in S_1$ by construction; at each step,
the UCB-based update (line~7) and Lipschitz continuity of~$g$
ensure $S_{n+1} \supseteq S_n \cap B_\epsilon(\beta_n)$
for sufficiently small~$\epsilon$, so the safe set never
collapses~\cite{BerkenkampSafeMPC}.
\item \emph{Lyapunov stability.}
The Lyapunov function $V_n = \tr(\Sigma_n)$ is strictly decreasing:
each rank-$1$ covariance contraction
(\cref{prop:kl-contract}) gives
$V_{n+1} = V_n - \sigma^2_{n-1}(\beta_n)\,\eta_n < V_n$.
Since $V_n \geq 0$ and $V_n \searrow$, the posterior concentrates;
$V_n \to 0$ iff the GP has resolved all uncertainty, i.e., the
cycle has closed.
\end{enumerate}
\end{remark}

\begin{remark}[Stein-variational iLQR]
\label{rmk:svgd-ilqr}
The GP-UCB acquisition (\cref{alg:recycle}, line~3) admits a
parallel, particle-based interpretation via Stein Variational
Gradient Descent~\cite{LiuSVGD}. Given $M$ particles
$\{\beta^{(m)}\}_{m=1}^M$, the SVGD update is
\[
\textcolor{PostColor}{
  \beta^{(m)} \gets \beta^{(m)}
  + \frac{\epsilon}{M}\sum_{j=1}^{M}\Bigl[
    \underbrace{k(\beta^{(j)}, \beta^{(m)})\,
      \nabla_{\!\beta^{(j)}} \log \pi(\beta^{(j)})}_{\text{drift (exploitation)}}
    + \underbrace{\nabla_{\!\beta^{(j)}}
      k(\beta^{(j)}, \beta^{(m)})}_{\text{repulsion (exploration)}}
  \Bigr],
}
\]
where $\pi(\beta) \propto \exp(-J(\beta)/\varepsilon)$.
The \emph{drift} term is the iLQR descent direction: $\nabla \log \pi$
derives from the Riccati cost-to-go, steering particles toward
low-cost gaits.
The \emph{repulsion} term is the kernel gradient, proportional to the
GP posterior variance $\sigma_{n-1}$, which pushes particles apart
to maintain exploration.
The connection to \cref{alg:recycle} is:
GP-UCB (sequential, global guarantee via $\gamma_T$) $\longleftrightarrow$
SVGD (parallel, local contraction via KL).
Both minimize the same KL objective; GP-UCB does so
greedily one point at a time, SVGD does so by transporting
the full particle ensemble.
\end{remark}

\begin{remark}[Kernel attention]
\label{rmk:attention}
The GP posterior mean is a kernel-weighted average:
\[
\textcolor{PostColor}{
  \mu_n(\beta)
  = \sum_{i=1}^{n} \alpha_i\, k(\beta, \beta_i)
  = \mathrm{Attn}\bigl(\beta;\, \{\beta_i\},\, \{J_i\}\bigr),
}
\]
where $\alpha = (K + \sigma^2_{\mathrm{noise}} I)^{-1} \mathbf{J}$
and $K_{ij} = k(\beta_i, \beta_j)$.
This is precisely the attention mechanism: the kernel $k$ plays
the role of the query-key similarity, the weights $\alpha_i$ are
the softmax scores, and the values are the observed costs~$J_i$.
The GP-LCB acquisition~\eqref{eq:gp-ucb} subtracts an
\emph{anti-attention} term:
\[
\textcolor{AcqColor}{
  \ell_n(\beta)
  = \underbrace{\mu_n(\beta)}_{\text{attention (exploit)}}
  - \sqrt{\bar\beta_n}\;
    \underbrace{\sigma_n(\beta)}_{\text{anti-attention (explore)}}.
}
\]
The posterior variance~$\sigma_n$ is maximal where the kernel
``attends least''---precisely the unexplored regions---so the
LCB selects where low expected cost overcomes exploration
uncertainty: known-good gaits minus the penalty for the unknown.
The SVGD interpretation (\cref{rmk:svgd-ilqr}) extends this:
$M$~particles constitute $M$~attention heads, each attending to the
score~$\nabla\log\pi$ (exploitation) while the repulsive kernel
gradient enforces head diversity (exploration).
\end{remark}

\begin{remark}[Goal-directed dynamics: closing the optimization-deployment gap]
\label{rem:gdd}
The Riccati feedback law $u = k(t) - K(t)(x - \bar{x}(t))$ produces optimal \emph{centroidal} foot forces, but deploying these on an articulated robot via the Jacobian-transpose mapping $\tau = J^\top f$ fails when the configuration drifts from the nominal trajectory.  The Jacobian $J(q)$ changes with $q$, so the effective wrench diverges from the intended one as tracking error accumulates.

\medskip\noindent\textbf{Resolution.}
Following Todorov's goal-directed dynamics framework~\cite{TodorovGDD,TodorovDSO}, we work in \emph{acceleration space} instead of force space.
Given the Riccati foot forces $f_k$ and contact schedule~$\sigma_k$:
\begin{enumerate}[label=(\roman*)]
\item Compute the desired centre-of-mass acceleration:
\[
\ddot{p}_{\mathrm{des}} = \frac{1}{m}\sum_k \sigma_k f_k + g, \qquad
\dot{\omega}_{\mathrm{des}} = I^{-1}\sum_k \sigma_k (r_k \times f_k).
\]
\item Set the generalized acceleration $\ddot{q}$ so that the base follows $(\ddot{p}_{\mathrm{des}},\, \dot{\omega}_{\mathrm{des}})$ and the joints track a PD target in acceleration space.
\item Call $\texttt{mj\_inverse}$ at the \emph{current} configuration $q$ to obtain the dynamically consistent joint torques $\tau = M(q)\ddot{q} + h(q,\dot{q})$.
\end{enumerate}
In acceleration space the mapping from desired motion to joint torques is kinematic---linear in $\ddot{q}$ regardless of configuration---so the controller remains accurate even when $q$ deviates from the nominal trajectory.
The Riccati gains serve as a \emph{value function approximation} for the cost evaluation (\cref{alg:riccati-eval}), while GDD provides the \emph{deployment} mechanism that realizes those forces on hardware.
\end{remark}

\begin{remark}[Per-sample domain randomization]
\label{rem:domain-rand}
Since each iteration of \cref{alg:recycle} evaluates its cost via \cref{alg:riccati-eval}, the cost function is a natural site for \emph{domain randomization}.
For each of the $M$ domain-randomized copies, we draw perturbed centroidal parameters from a prior distribution:
\[
m^{(i)} \sim m(1 + \delta_m),\quad
I^{(i)} \sim I \odot (1 + \delta_I),\quad
r_k^{(i)} \sim r_k + \delta_r,
\]
where $\delta_m \sim \cN(0, \sigma_m^2)$, $\delta_I \sim \cN(0, \sigma_I^2 \mathbb{1}_3)$, and $\delta_r \sim \cN(0, \sigma_r^2 \mathbb{1}_3)$.
Each copy's Riccati solve (\cref{alg:riccati-eval}, line~2) and forward simulation (line~3) use the perturbed dynamics, so the GP posterior update (lines~5--6 of \cref{alg:recycle}) automatically learns gait parameters $\beta$ that achieve low cost \emph{across} the uncertainty set.
This is a form of distributionally robust optimization: the Gaussian mixture induced by the $M$ parameter perturbations defines an empirical uncertainty set, and the GP acquisition finds gaits robust to worst-case-in-expectation model error---at negligible additional cost, since the per-sample Riccati solve dominates the computational budget.
The domain-randomized cost contributes observation noise $\sigma^2_{\mathrm{noise}}$ to the GP likelihood, which the posterior update (line~6) absorbs naturally: larger randomization increases $\sigma^2_{\mathrm{noise}}$, widening the posterior confidence bands and promoting exploration.
\end{remark}

\begin{remark}[Constrained unscented GP]
\label{rmk:unscented}
The unscented transform connects the three algorithmic levels---inner
trajectory solve, noise averaging, and outer Bayesian
optimization---into a single deterministic framework.

\smallskip\noindent\emph{Level~1: derivative-free backward pass
(\cref{alg:riccati-eval}, line~2).}
The linearization $\text{dynamics} \to A(t), B(t)$ is vulnerable to
sharp contact transitions: as $\varepsilon \to 0$,
$\sigma_k(t)$ switches rapidly and the Riccati gains can
diverge---the standard BPTT failure mode.
Unscented dynamic programming (UDP)~\cite{ManchesterUDP} replaces
linearization with $2d_x{+}1$ sigma points propagated through the
\emph{full nonlinear} dynamics, fitting the cost-to-go quadratic
directly from sigma-point statistics.
No Jacobians are computed; the dynamics is a black box.
Constrained UDP (CUDP)~\cite{PlancherCUDP} adds an augmented
Lagrangian for state and control constraints, ensuring
\emph{backward feasibility}: the backward pass produces gains
$K(t)$ that keep the forward trajectory in the feasible set.

\smallskip\noindent\emph{Level~2: deterministic noise averaging
(\cref{alg:riccati-eval}, lines~3--4).}
The domain-randomized copies (\cref{rem:domain-rand}) are
replaced by $2p{+}1$ sigma points in the centroidal
parameter space $\theta = (m, I, r_1, \ldots, r_4) \in \R^p$:
\[
  J_n = \sum_{j=0}^{2p} w^{(j)}\, J(\beta_n;\, \theta^{(j)}),
\]
exact for Gaussian $\theta$ through a quadratic cost landscape.
For a quadruped ($p \approx 16$), $33$ deterministic
evaluations replace $M$ random rollouts.

\smallskip\noindent\emph{Level~3: deterministic GP
(\cref{alg:recycle}).}
The GP posterior update is a Kalman filter step in function space:
the measurement model $y_n = J(\beta_n) + \text{noise}$ is
\emph{linear} in the unknown function~$J$, so the Kalman update
is exact---no unscented approximation is needed at this level.
But the converse holds: the GP posterior is \emph{recoverable}
from sigma-point evaluations.
Choose $2d{+}1$ sigma points
$\{\beta^{(j)}\}$ from the GP prior covariance
in $\bT^d$ ($d = \dim\Theta$; for a quadruped with
$\Theta = \bT^3$, this is $7$~points).
Evaluate $J(\beta^{(j)})$ at each via \cref{alg:riccati-eval}.
The GP posterior conditioned on these $2d{+}1$ evaluations is:
\[
\textcolor{PostColor}{
  \mu_{2d+1}(\beta)
  = \mathbf{k}(\beta)^\top\,
    (K + \sigma^2_{\mathrm{noise}} I)^{-1}\, \mathbf{J},
}
\]
where $\mathbf{k}(\beta)_j = k(\beta, \beta^{(j)})$ and
$K_{ij} = k(\beta^{(i)}, \beta^{(j)})$.
The sigma points provide the optimal query locations for
this batch: they capture the prior covariance structure,
so $\mu_{2d+1}$ is the best $2d{+}1$-point GP
approximation.
This is a \emph{deterministic GP}: $2d{+}1$ function evaluations,
no random sampling, no sequential acquisition---one batch
produces the full posterior.
\cref{alg:recycle} implements this directly:
Phase~1 (line~1, iterations $n \leq 2d{+}1$) evaluates the
sigma-point batch; Phase~2 ($n > 2d{+}1$) refines via GP-UCB,
selecting additional queries guided by the posterior
variance and converging at the information-theoretic
rate~$\gamma_T$.

\smallskip\noindent\emph{Feasibility unification.}
The two feasibility constraints are the same mechanism at
different scales:
\begin{enumerate}[label=(\roman*),nosep]
\item \emph{inner} (CUDP): backward feasibility---gains $K(t)$
  keep the trajectory $x(t)$ in the constraint set;
\item \emph{outer} (safe GP-UCB): recursive feasibility---the
  safe set $S_n$ never collapses (\cref{rmk:mpc}).
\end{enumerate}
Both are enforced by constrained backward propagation at their
respective scales: augmented Lagrangian for the inner dynamics,
lower confidence bound for the outer gait space.
\end{remark}

\begin{remark}[Measure-theoretic unification:
$\sigma$-algebra, attention, and the unscented transform]
\label{rmk:sigma-algebra}
The preceding remarks---kernel attention (\cref{rmk:attention}),
MPC interpretation (\cref{rmk:mpc}), constrained unscented GP
(\cref{rmk:unscented}), and Stein-variational iLQR
(\cref{rmk:svgd-ilqr})---are manifestations of a single
measure-theoretic structure: the filtration generated by
GP observations.

\smallskip\noindent\emph{The observation filtration.}
Let $\cF_n = \sigma(\mathcal{D}_n)
= \sigma(\beta_1, J_1, \ldots, \beta_n, J_n)$ be the
$\sigma$-algebra generated by the first~$n$ observations.
The sequence $\cF_0 \subset \cF_1 \subset \cdots \subset \cF_T$
is a filtration: each iteration of \cref{alg:recycle} refines
$\cF_n$ by exactly one observation $(\beta_{n+1}, J_{n+1})$.
The algorithm is $\cF_n$-adapted: the acquisition
$\beta_{n+1}$ is a deterministic function of $\cF_n$.

\smallskip\noindent\emph{Conditional expectation as GP posterior.}
For Gaussian processes, the conditional expectation and
posterior mean coincide by definition:
\[
  \textcolor{PostColor}{
  \mathbb{E}[J(\beta) \mid \cF_n] = \mu_n(\beta), \qquad
  \Var(J(\beta) \mid \cF_n) = \sigma_n^2(\beta).
  }
\]
This is not an approximation but an identity:
$\mathbb{E}[\,\cdot \mid \cF_n]$ is the $L^2$-orthogonal
projection onto the subspace of $\cF_n$-measurable functions.

\smallskip\noindent\emph{Attention as projection.}
The identification $\mu_n = \mathbb{E}[\,\cdot \mid \cF_n]$
gives the attention--anti-attention decomposition of
\cref{rmk:attention} a precise operator-theoretic meaning:
\begin{enumerate}[label=(\roman*),nosep]
\item \emph{attention} $= \mathbb{E}[J(\beta) \mid \cF_n]
  = \mu_n(\beta)$: projection onto ``what is known'';
\item \emph{anti-attention}
  $= \sqrt{\Var(J(\beta) \mid \cF_n)} = \sigma_n(\beta)$:
  norm of the residual in $\cF_n^\perp$, i.e.,
  ``what is unknown'';
\item \emph{LCB} $= \mathbb{E}[J \mid \cF_n]
  - \sqrt{\bar\beta_n}\;\|J - \mathbb{E}[J \mid \cF_n]\|
  $: optimistic projection for cost, known minus calibrated unknown.
\end{enumerate}

\smallskip\noindent\emph{Sigma points as optimal initialization.}
For a Gaussian prior over $\Theta \subseteq \R^d$, the
$2d{+}1$ sigma points are the \emph{optimal batch design}
for recovering the first two moments of any function
pushed through the prior:
\[
  \textcolor{CycleColor}{
  \cF_{2d+1}^{\mathrm{UT}}
  = \sigma\!\bigl(J(\beta^{(1)}), \ldots, J(\beta^{(2d+1)})\bigr),
  \qquad
  \mu_{2d+1}(\beta)
  = \mathbb{E}[J(\beta) \mid \cF_{2d+1}^{\mathrm{UT}}],
  }
\]
where $\mu_{2d+1}$ is the GP posterior conditioned on these
$2d{+}1$ observations (Level~3 of \cref{rmk:unscented}).
This is a finite-dimensional approximation, not
function-space sufficiency: $V_{2d+2} < V_{2d+1}$
(\cref{rmk:mpc}), so further observations always improve
the posterior.
Phase~1 of \cref{alg:recycle} constructs
$\cF_{2d+1}^{\mathrm{UT}}$ as the best batch initialization;
Phase~2 refines it adaptively:
$\cF_{2d+1} \subset \cF_{2d+2} \subset \cdots \subset \cF_T$.

\smallskip\noindent\emph{MPC as filtration dynamics.}
The MPC interpretation (\cref{rmk:mpc}) becomes:
the state $s_n = (\mu_n, \Sigma_n)$ is the conditional law
of~$J$ given~$\cF_n$.
The Lyapunov function $V_n = \tr(\Sigma_n)$ measures the
``size'' of~$\cF_n^\perp$---the unexplained variance.
Each observation contracts $\cF_n^\perp$ by a rank-$1$
projection, giving $V_{n+1} < V_n$.
The cycle closes when $\cF_T = \cF_\infty$, i.e., all
uncertainty has been resolved and the holonomy vanishes.

\smallskip\noindent\emph{Randomness structure.}
Two disjoint $\sigma$-algebras carry distinct sources of
randomness:
\begin{enumerate}[label=(\roman*),nosep]
\item \emph{Injective randomness.}
  At each iteration~$n$, fresh domain-randomized parameters
  $\mathcal{G}_n = \sigma(\theta_n^{(1)}, \ldots, \theta_n^{(M)})$
  (\cref{rem:domain-rand}) are drawn independently of the
  observation history~$\cF_{n-1}$.
  Their effect is absorbed into $\sigma^2_{\mathrm{noise}}$
  in the GP likelihood: the noise realization affects~$J_n$
  (and hence enters~$\cF_n$), but the acquisition
  $\beta_{n+1}$ was determined from~$\cF_n$ before
  $\mathcal{G}_{n+1}$ is realized.
\item \emph{Inductive randomness}: \textbf{none.}
  The acquisition $\beta_{n+1}$ is a deterministic function
  of~$\cF_n$ (Phase~1: precomputed sigma points; Phase~2:
  $\arg\min$ of LCB).
  The algorithm has zero inductive randomness---every decision
  is a measurable function of the observation filtration.
  This is a categorical distinction from MPPI, where the
  sampling distribution introduces inductive randomness that
  cannot be replayed deterministically.
\end{enumerate}
\end{remark}
